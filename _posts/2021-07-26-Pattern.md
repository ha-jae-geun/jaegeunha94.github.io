# 프로토타입 프로퍼티 제거 예제
```javascript
function Person() {}

Person.prototype.name = 'jaegeun'

var person1 = new Person();
var person2 = new Person();

person1.name = "jg"

alert(person1.name)

delete person1.name;

alert(person1.name)

```
